normalize_identifier(identifier)

normalizes an OpenID according to openid.net/specs/openid-authentication-2_0.html#normalization

 77:   def self.normalize_identifier(identifier)
 78:     # clean up whitespace
 79:     identifier = identifier.to_s.strip
 80: 
 81:     # if an XRI has a prefix, strip it.
 82:     identifier.gsub!(/xri:\/\//i, '')
 83: 
 84:     # dodge XRIs -- TODO: validate, don't just skip.
 85:     unless ['=', '@', '+', '$', '!', '('].include?(identifier.at(0))
 86:       # does it begin with http?  if not, add it.
 87:       identifier = "http://#{identifier}" unless identifier =~ /^http/i
 88: 
 89:       # strip any fragments
 90:       identifier.gsub!(/\#(.*)$/, '')
 91: 
 92:       begin
 93:         uri = URI.parse(identifier)
 94:         uri.scheme = uri.scheme.downcase  # URI should do this
 95:         identifier = uri.normalize.to_s
 96:       rescue URI::InvalidURIError
 97:         raise InvalidOpenId.new("#{identifier} is not an OpenID identifier")
 98:       end
 99:     end
100: 
101:     return identifier
102:   end
